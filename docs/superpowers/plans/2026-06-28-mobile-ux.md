# 移动端体验优化实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 让 OmniTerm 在纯触屏手机上可完整使用，改造移动端布局、导航、终端、文件管理器的触控体验，同时保持桌面端零回归。

**Architecture:** 以 `Layout.tsx` 的 `isMobile` 分支为切入点，拆出 `MobileStatusBar` 和 `MobileNav` 两个独立组件；`appStore` 新增移动端状态；终端侧通过 `MobileKeyBar` 和 xterm.js 触摸/滚动模式解决软键盘与滚动问题；文件管理器通过样式覆盖和长按菜单适配触屏。

**Tech Stack:** React 19 / Vite 8 / TailwindCSS 4 / Zustand / @xterm/xterm / vitest

## Global Constraints

- 断点保持 `max-width: 768px`
- 禁止在 UI 中使用 emoji（`docs/ui-style-guide.md` §5.0）
- 图标必须为 SVG / `currentColor` / 18×18 / viewBox `0 0 16 16`
- 按钮触控区最小 40×40
- 所有改动限制在移动端分支或独立组件，桌面端 untouched
- 每完成一个可测试任务后立即 `git commit`
- 变更日志 `CHANGELOG.md` 仅写用户确认的内容（本期先不写）

- [ ] **Step 3: 修改 Layout.tsx 的移动端分支**

保留桌面端逻辑不变，将移动端分支改为：

```tsx
import { MobileStatusBar } from './MobileStatusBar'
import { MobileNav } from './MobileNav'
// ... existing imports

function MobileContent() {
  const activeTab = useAppStore((s) => s.activeTab)
  const activeSessionId = useAppStore((s) => s.activeSessionId)

  switch (activeTab) {
    case 'terminal':
      return <Terminal key={activeSessionId ?? 'empty'} />
    case 'files':
      return <FileManager />
    case 'sessions':
      return <Sidebar />
    default:
      return <Terminal key={activeSessionId ?? 'empty'} />
  }
}

// 在 Layout 组件内，isMobile 分支前添加手势 hook
function useMobileGestures() {
  const { activeTab, setActiveTab, mobileGestureEnabled } = useAppStore()

  const handleSwipe = useCallback((direction: 'left' | 'right') => {
    const order: AppState['activeTab'][] = ['sessions', 'terminal', 'files']
    const idx = order.indexOf(activeTab)
    if (idx === -1) return
    const next = direction === 'left' ? idx + 1 : idx - 1
    if (next >= 0 && next < order.length) {
      setActiveTab(order[next])
    }
  }, [activeTab, setActiveTab])

  return { handleSwipe, enabled: mobileGestureEnabled }
}

// 在 isMobile 分支中使用
if (isMobile) {
  const { handleSwipe, enabled } = useMobileGestures()
  const activeSessionName = sessions.find((s) => s.id === activeSessionId)?.name ?? activeSessionId ?? t('sidebar.noSessions')

  return (
    <div
      className="flex flex-col h-screen"
      style={{ background: 'var(--bg-base)', color: 'var(--text-primary)', overflow: 'hidden' }}
    >
      <MobileStatusBar
        connected={connected}
        sessionName={activeSessionName}
        onSessionClick={() => setActiveTab('sessions')}
        onNewSession={() => setActiveTab('sessions') /* 进入会话 tab 后点击新建 */}
      />
      <div
        className="flex-1 overflow-hidden"
        onTouchStart={enabled ? (e) => {
          const touch = e.touches[0]
          ;(e.currentTarget as HTMLDivElement).dataset.startX = String(touch.clientX)
          ;(e.currentTarget as HTMLDivElement).dataset.startY = String(touch.clientY)
        } : undefined}
        onTouchEnd={enabled ? (e) => {
          const div = e.currentTarget as HTMLDivElement
          const startX = parseFloat(div.dataset.startX ?? '0')
          const startY = parseFloat(div.dataset.startY ?? '0')
          const touch = e.changedTouches[0]
          const dx = touch.clientX - startX
          const dy = touch.clientY - startY
          const edgeMargin = 24
          if (Math.abs(dx) < Math.abs(dy)) return
          if (Math.abs(dx) < 40) return
          // 避开两侧边缘，防止触发系统返回
          if (startX < edgeMargin || startX > window.innerWidth - edgeMargin) return
          handleSwipe(dx < 0 ? 'left' : 'right')
        } : undefined}
      >
        <MobileContent />
      </div>
      <MobileNav />
    </div>
  )
}
```

- [ ] **Step 4: 修改 Terminal.tsx**

```tsx
import { useRef, useEffect, useState } from 'react'
import { useTranslation } from 'react-i18next'
import { useAppStore } from '../../stores/appStore'
import { useTerminal } from '../../hooks/useTerminal'
import { KeyboardIcon } from '../Icons/KeyboardIcon'
import { MobileKeyBar } from './MobileKeyBar'

const FONT = "'JetBrains Mono', 'Fira Code', 'Cascadia Code', ui-monospace, monospace"

export function Terminal() {
  const { t } = useTranslation()
  const containerRef = useRef<HTMLDivElement>(null)
  const activeSessionId = useAppStore((s) => s.activeSessionId)
  const isMobile = useAppStore((s) => s.isMobile)
  const fontSize = useAppStore((s) => s.fontSize)
  const mobileFontSize = useAppStore((s) => s.mobileFontSize)
  const effectiveFontSize = isMobile ? mobileFontSize : fontSize
  const { initTerminal, sendData, scrollMode, setScrollMode, sendScrollKeys, exitScrollMode } = useTerminal({
    sessionId: activeSessionId,
    fontSize: effectiveFontSize,
  })

  useEffect(() => {
    if (containerRef.current) {
      const cleanup = initTerminal(containerRef.current)
      return cleanup
    }
  }, [initTerminal])

  const handleKey = (name: string) => {
    if (!sendData) return
    switch (name) {
      case 'Ctrl':
        // 组合模式通过后续按键处理，此处简单发送 \x00 占位
        sendData('\x00')
        break
      case 'Esc':
        sendData('\x1b')
        if (scrollMode) exitScrollMode?.()
        break
      case 'Tab':
        sendData('\t')
        break
      case '←':
        sendData('\x1b[D')
        break
      case '→':
        sendData('\x1b[C')
        break
      case '↑':
        isMobile && scrollMode ? sendScrollKeys?.('up') : sendData('\x1b[A')
        break
      case '↓':
        isMobile && scrollMode ? sendScrollKeys?.('down') : sendData('\x1b[B')
        break
      case '复制':
        // 由外部调用 navigator.clipboard，此处触发 toast 在 hook 中处理
        break
      case '粘贴':
        navigator.clipboard.readText().then((text) => sendData(text)).catch(() => {})
        break
    }
  }

  if (!activeSessionId) {
    return (
      <div className="h-full flex items-center justify-center" style={{ background: 'var(--bg-base)', color: 'var(--text-faint)', fontFamily: FONT }}>
        {/* ... existing empty state ... */}
      </div>
    )
  }

  return (
    <div style={{ height: '100%', display: 'flex', flexDirection: 'column', background: 'var(--bg-base)' }}>
      <div ref={containerRef} className="h-full w-full p-1" style={{ flex: 1, minHeight: 0 }} />
      {isMobile && (
        <MobileKeyBar
          onKey={handleKey}
          scrollMode={scrollMode ?? false}
          onToggleScrollMode={() => {
            if (scrollMode) {
              exitScrollMode?.()
            } else {
              setScrollMode?.(true)
            }
          }}
        />
      )}
    </div>
  )
}
```
